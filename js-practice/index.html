③ 表达式

// 算术、比较、逻辑、三元表达式
let a = 10, b = 3;

console.log("算术: ", a + b, a - b, a * b, a / b, a % b, a ** b);
console.log("比较: ", a > b, a === b, a !== b);
console.log("逻辑: ", a > 5 && b < 5, a < 5 || b < 5, !false);

④ 流程控制

// if / else if / else
let score = 75;
if (score >= 90)      console.log("优秀");
else if (score >= 60) console.log("及格");
else                  console.log("不及格");

// switch
let day = 3;
switch (day) {
  case 1: console.log("周一"); break;
  case 3: console.log("周三"); break;
  default: console.log("其他");
}

// for 循环
let sum = 0;
for (let i = 1; i <= 5; i++) sum += i;
console.log("1~5 之和: ", sum);

// while 循环
let n = 1, result = 1;
while (n <= 5) { result *= n; n++; }
console.log("5! = ", result);

// for...of 遍历数组
let fruits = ["苹果", "香蕉", "橙子"];
for (const f of fruits) console.log("水果: ", f);

⑤ 函数

// 函数声明
function greet(name) {
  return `Hello, ${name}!`;
}
console.log(greet("世界"));

// 函数表达式
const square = function(x) { return x * x; };
console.log("3 的平方: ", square(3));

// 箭头函数
const add = (a, b) => a + b;
console.log("箭头函数 5+7 = ", add(5, 7));

// 默认参数
function power(base, exp = 2) {
  return base ** exp;
}
console.log("power(4) = ", power(4));
console.log("power(2, 10) = ", power(2, 10));

// 递归函数
function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}
console.log("fibonacci(8) = ", fibonacci(8));

// 高阶函数
const nums = [1, 2, 3, 4, 5];
const doubled = nums.map(x => x * 2);
console.log("map 结果: ", doubled.join(", "));
